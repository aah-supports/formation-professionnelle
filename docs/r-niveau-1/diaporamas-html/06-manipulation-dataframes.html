Filtrage de lignes

Garder ce qui compte

# Condition simple
mtcars[mtcars$cyl == 6, ]

# Conditions combinées
mtcars[mtcars$cyl > 4 & mtcars$hp > 100, ]

# Avec subset (plus lisible)
subset(mtcars, cyl > 4 & hp > 100,
       select = c(mpg, cyl, hp))

Colonnes calculées

Enrichir vos données

df <- mtcars[1:5, c("mpg", "wt", "hp")]

# Nouvelle colonne
df$ratio <- df$hp / df$wt

# Avec transform
df <- transform(df,
  efficacite = mpg / hp,
  categorie  = ifelse(mpg > 20, "eco", "sport")

Merge et jointures

Fusionner deux tables

produits <- data.frame(
  id = 1:3,
  nom = c("Pomme", "Pain", "Lait")
)
prix <- data.frame(
  id = c(1, 2, 4),
  prix = c(1.5, 2.0, 3.0)
)

# Inner join (défaut)
merge(produits, prix, by = "id")

# Left join
merge(produits, prix, by = "id", all.x = TRUE)

# Full join
merge(produits, prix, by = "id", all = TRUE)

Agrégation

Résumer par groupe

# Moyenne de mpg par cylindres
aggregate(mpg ~ cyl, data = mtcars, FUN = mean)
#   cyl      mpg
# 1   4 26.66364
# 2   6 19.74286
# 3   8 15.10000

# Plusieurs stats
aggregate(cbind(mpg, hp) ~ cyl,
  data = mtcars, FUN = mean)

# Avec tapply
tapply(mtcars$mpg, mtcars$cyl, mean)
