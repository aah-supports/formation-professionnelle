plot() — Nuage de points

La fonction graphique universelle

plot(mtcars$wt, mtcars$mpg,
  main = "Poids vs Consommation",
  xlab = "Poids (1000 lbs)",
  ylab = "Miles par gallon",
  col  = "steelblue",
  pch  = 19,   # point plein
  cex  = 1.2   # taille
)
abline(lm(mpg ~ wt, data = mtcars),
  col = "red", lwd = 2)

hist() — Histogramme

La distribution d'une variable

hist(mtcars$mpg,
  breaks = 10,
  col    = "#a78bfa",
  border = "white",
  main   = "Distribution de mpg",
  xlab   = "Miles par gallon"
)

# Ajouter la courbe de densité
lines(density(mtcars$mpg),
  col = "red", lwd = 2)

boxplot() — Boîte à moustaches

Comparer des groupes

boxplot(mpg ~ cyl, data = mtcars,
  main = "Consommation par cylindres",
  xlab = "Nombre de cylindres",
  ylab = "Miles par gallon",

barplot() — Diagramme en barres

Compter et comparer

# Compter par catégorie
freqs <- table(mtcars$cyl)

barplot(freqs,
  main = "Nombre de voitures par cylindres",
  xlab = "Cylindres",
  ylab = "Nombre",
  col  = c("#00f3ff", "#a78bfa", "#ff6b00")
)

# Ajouter une légende
legend("topright", legend = names(freqs),
  fill = c("#00f3ff", "#a78bfa", "#ff6b00"))

Combiner des graphiques

Dashboard en une commande

par(mfrow = c(2, 2))

plot(mtcars$wt, mtcars$mpg, main = "Poids vs MPG")
hist(mtcars$mpg, main = "Distribution MPG", col = "#a78bfa")
boxplot(mpg ~ cyl, data = mtcars, main = "MPG par Cyl")
barplot(table(mtcars$gear), main = "Vitesses", col = "#10b981")

par(mfrow = c(1, 1))  # remettre à 1
Disposition 2 x 2

Exporter un graphique

PDF, PNG et plus

# Exporter en PNG
png("mon_graphique.png",
  width = 800, height = 600, res = 150)
plot(mtcars$wt, mtcars$mpg,
  main = "Poids vs Consommation",
  col = "steelblue", pch = 19)
dev.off()

# Exporter en PDF
pdf("mon_graphique.pdf", width = 8, height = 6)
hist(mtcars$mpg, col = "#a78bfa")
dev.off()
