Fixtures pytest réutilisables

import pytest

@pytest.fixture
def sample_orders():
    return [
        {"id": 1, "amount": 120.0},
        {"id": 2, "amount": 80.0},
    ]

def test_total_amount(sample_orders):
    total = sum(o["amount"] for o in sample_orders)
    assert total == 200.0

Paramétrage pour cas limites

import pytest

@pytest.mark.parametrize(
    "value,expected",
    [(0, "low"), (300, "medium"), (1000, "high")],
)
def test_bucketize(value, expected):
    assert bucketize(value) == expected
  • Lisible et compact pour plusieurs scénarios
  • Évite la duplication de fonctions de tests

Mocking des dépendances externes

from unittest.mock import Mock

def test_sync_calls_gateway_once():
    gateway = Mock()
    service = SyncService(gateway=gateway)

    service.sync([{"id": "A1"}])

    gateway.push.assert_called_once_with({"id": "A1"})

Tests async et timeouts

import pytest

@pytest.mark.asyncio
async def test_fetch_kpi_timeout(client):
    result = await client.fetch_kpi(timeout=0.2)
    assert result.status == "ok"
  • Toujours encadrer les appels réseau par timeout
  • Vérifier les chemins d'erreur, pas seulement le chemin heureux
