Étape 1 — Extraction

con <- DBI::dbConnect(RSQLite::SQLite(), "business.db")
ventes_tbl <- dplyr::tbl(con, "ventes")
clients_tbl <- dplyr::tbl(con, "clients")

catalogue <- readxl::read_excel("catalogue.xlsx", sheet = "Produits")
objectifs <- readxl::read_excel("objectifs.xlsx", sheet = "Objectifs", skip = 2)

Étape 3 — Table analytique

pilotage <- ventes_tbl |>
  left_join(clients_tbl, by = "client_id") |>
  collect() |>
  left_join(catalogue, by = "produit_id") |>
  mutate(montant_commande = quantite * prix_net) |>
  group_by(mois, region, segment) |>
  summarise(ca = sum(montant_commande, na.rm = TRUE), .groups = "drop") |>
  left_join(objectifs, by = c("mois", "region"))
