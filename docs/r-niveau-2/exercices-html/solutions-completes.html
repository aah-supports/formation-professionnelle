Module 3 : Jointures et requêtes avancées

clients_tbl <- tbl(con, "clients")

base_analyse <- ventes_tbl |>
  left_join(clients_tbl, by = "client_id") |>
  collect() |>
  left_join(catalogue, by = "produit_id") |>
  mutate(montant_commande = quantite * prix_net)

# Produits vendus hors catalogue
produits_absents <- base_analyse |>
  distinct(produit_id) |>
  anti_join(catalogue |> distinct(produit_id), by = "produit_id")

# Top clients par région
top_clients <- base_analyse |>
  group_by(region, client_id) |>
  summarise(ca = sum(montant_commande, na.rm = TRUE), .groups = "drop") |>
  group_by(region) |>
  mutate(rang = min_rank(desc(ca))) |>
  filter(rang <= 5) |>
  arrange(region, rang)

Module 4 : Fenêtres et automatisation

library(purrr)

# Évolution mensuelle
evo_mensuelle <- base_analyse |>
  mutate(mois = as.Date(format(date, "%Y-%m-01"))) |>
  group_by(region, mois) |>
  summarise(ca = sum(montant_commande, na.rm = TRUE), .groups = "drop") |>
  group_by(region) |>
  arrange(mois, .by_group = TRUE) |>
  mutate(
    ca_mois_prec = lag(ca),
    var_mom = (ca - ca_mois_prec) / ca_mois_prec
  ) |>
  ungroup()

# Fonction paramétrée
extraire_kpi <- function(con, date_debut, date_fin, region = NULL) {
  req <- tbl(con, "ventes") |>
    filter(date >= date_debut, date <= date_fin)

  if (!is.null(region)) req <- req |> filter(region == !!region)

  req |>
    summarise(
      ca_total = sum(prix_net, na.rm = TRUE),
      nb_commandes = n_distinct(id_commande),
      panier_moyen = ca_total / nb_commandes
    ) |>
    collect()
}

Module 5 : Contrôles qualité

# Profil NA et cardinalité
profil_qc <- base_analyse |>
  summarise(across(
    everything(),

  ))

# Doublons de clé commande/date
doublons <- base_analyse |>
  count(id_commande, date, name = "n") |>
  filter(n > 1)

# Réconciliation base vs Excel
ca_base <- base_analyse |>
  mutate(mois = as.Date(format(date, "%Y-%m-01"))) |>
  group_by(mois, region) |>
  summarise(ca_base = sum(montant_commande, na.rm = TRUE), .groups = "drop")

reco <- ca_base |>
  full_join(objectifs |> rename(ca_excel = objectif_ca), by = c("mois", "region")) |>
  mutate(
    ca_base = coalesce(ca_base, 0),
    ca_excel = coalesce(ca_excel, 0),
    ecart = ca_base - ca_excel,
    ecart_pct = if_else(ca_excel == 0, NA_real_, ecart / ca_excel)
  )

Module 6 : Statistiques descriptives pratiques

# Indicateurs globaux
kpi_global <- base_analyse |>
  summarise(
    ca_total = sum(montant_commande, na.rm = TRUE),
    panier_moyen = mean(montant_commande, na.rm = TRUE),
    panier_mediane = median(montant_commande, na.rm = TRUE),
    iqr_panier = IQR(montant_commande, na.rm = TRUE),
    nb_clients = n_distinct(client_id)
  )

# Analyse segment
desc_segment <- base_analyse |>
  group_by(segment) |>
  summarise(
    nb_clients = n_distinct(client_id),
    ca_total = sum(montant_commande, na.rm = TRUE),
    panier_moyen = mean(montant_commande, na.rm = TRUE),
    panier_mediane = median(montant_commande, na.rm = TRUE),
    iqr_panier = IQR(montant_commande, na.rm = TRUE),
    .groups = "drop"
  ) |>
  mutate(part_ca = ca_total / sum(ca_total)) |>
  arrange(desc(ca_total))

# Série temporelle
serie_mensuelle <- base_analyse |>
  mutate(mois = as.Date(format(date, "%Y-%m-01"))) |>
  group_by(mois) |>
  summarise(ca = sum(montant_commande, na.rm = TRUE), .groups = "drop") |>
  arrange(mois) |>
  mutate(
    ca_mois_prec = lag(ca),
    var_mom = (ca - ca_mois_prec) / ca_mois_prec,
    cumul_annuel = cumsum(ca),
    mm3 = (lag(ca, 2) + lag(ca, 1) + ca) / 3
  )
