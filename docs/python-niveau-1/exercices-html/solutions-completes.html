Exercice 3.4 : Fonctions simples

def celsius_to_fahrenheit(c):
    return c * 9 / 5 + 32

def imc(poids, taille):
    return round(poids / taille ** 2, 2)

def est_premier(n):
    if n < 2: return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0: return False
    return True

print(celsius_to_fahrenheit(0), celsius_to_fahrenheit(100))
print(imc(75, 1.80))
print([n for n in range(1, 21) if est_premier(n)])
