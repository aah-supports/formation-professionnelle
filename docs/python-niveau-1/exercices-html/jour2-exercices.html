Exercice 5.1 : Lire un fichier texte [Difficulté : Facile]

Énoncé :

  1. Creez un fichier villes.txt contenant 5 villes (une par ligne) en utilisant open() en mode ecriture
  2. Relisez le fichier avec open() et read() pour obtenir tout le contenu
  3. Relisez le fichier avec readlines() pour obtenir une liste de lignes
  4. Relisez le fichier avec une boucle for line in f
  5. Utilisez le mot-cle with pour une gestion propre du fichier

Indice : open("fichier", "w") ouvre en ecriture, open("fichier", "r") en lecture. Le bloc with open(...) as f: ferme automatiquement le fichier.

Solution 
# 1. Creer le fichier
with open("villes.txt", "w") as f:
    f.write("Paris\n")
    f.write("Lyon\n")
    f.write("Marseille\n")
    f.write("Toulouse\n")
    f.write("Bordeaux\n")

# 2. Lire tout le contenu
with open("villes.txt", "r") as f:
    contenu = f.read()
print("--- read() ---")
print(contenu)

# 3. Lire en liste de lignes
with open("villes.txt", "r") as f:
    lignes = f.readlines()
print(f"readlines() : {lignes}")

# 4. Boucle ligne par ligne
print("--- Boucle for ---")
with open("villes.txt", "r") as f:
    for ligne in f:
        print(ligne.strip())

# 5. Deja fait avec with ci-dessus
# Version nettoyee :
with open("villes.txt", "r") as f:
    villes = [ligne.strip() for ligne in f]
print(f"\nListe nettoyee : {villes}")

Exercice 5.2 : Écrire dans un fichier [Difficulté : Facile]

Énoncé :

  1. Ecrivez une liste de courses dans un fichier courses.txt (mode "w")
  2. Ajoutez 2 articles supplementaires avec le mode append ("a")
  3. Relisez le fichier et affichez le contenu avec les numeros de ligne
  4. Ecrivez une liste de nombres (1 a 10) dans un fichier, un par ligne, avec writelines()

Indice : Le mode "w" ecrase le fichier existant. Le mode "a" ajoute a la fin. writelines() n'ajoute pas de retour a la ligne automatiquement.

Solution 
# 1. Ecrire la liste de courses
articles = ["pain", "lait", "beurre", "oeufs", "fromage"]
with open("courses.txt", "w") as f:
    for article in articles:
        f.write(article + "\n")

# 2. Ajouter en mode append
with open("courses.txt", "a") as f:
    f.write("pommes\n")
    f.write("chocolat\n")

# 3. Relire avec numeros de ligne
print("--- Liste de courses ---")
with open("courses.txt", "r") as f:
    for i, ligne in enumerate(f, start=1):
        print(f"{i}. {ligne.strip()}")

# 4. writelines()
nombres = [f"{n}\n" for n in range(1, 11)]
with open("nombres.txt", "w") as f:
    f.writelines(nombres)

with open("nombres.txt", "r") as f:
    print(f"\nnombres.txt : {f.read()}")

Exercice 5.3 : Manipuler des CSV [Difficulté : Intermédiaire]

Énoncé :

  1. Creez un fichier employes.csv avec les colonnes : nom, poste, salaire (5 lignes)
  2. Relisez le fichier avec csv.reader et affichez chaque ligne
  3. Relisez le fichier avec csv.DictReader et affichez les noms et salaires
  4. Ajoutez un employe avec csv.writer
  5. Calculez le salaire moyen a partir du fichier

Indice : import csv. csv.reader(f) retourne un iterateur de listes. csv.DictReader(f) retourne un iterateur de dictionnaires. csv.writer(f).writerow(liste) ecrit une ligne.

Solution 
import csv

# 1. Creer le CSV
with open("employes.csv", "w", newline="") as f:
    writer = csv.writer(f)
    writer.writerow(["nom", "poste", "salaire"])
    writer.writerow(["Dupont", "Developpeur", 45000])
    writer.writerow(["Martin", "Designer", 40000])
    writer.writerow(["Bernard", "Chef de projet", 52000])
    writer.writerow(["Petit", "Analyste", 42000])
    writer.writerow(["Robert", "Developpeur", 47000])

# 2. Lire avec csv.reader
print("--- csv.reader ---")
with open("employes.csv", "r") as f:
    reader = csv.reader(f)
    for ligne in reader:
        print(ligne)

# 3. Lire avec DictReader
print("\n--- csv.DictReader ---")
with open("employes.csv", "r") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(f"{row['nom']} : {row['salaire']} EUR")

# 4. Ajouter un employe
with open("employes.csv", "a", newline="") as f:
    writer = csv.writer(f)
    writer.writerow(["Leroy", "Stagiaire", 28000])

# 5. Salaire moyen
with open("employes.csv", "r") as f:
    reader = csv.DictReader(f)
    salaires = [int(row["salaire"]) for row in reader]

moyenne = sum(salaires) / len(salaires)
print(f"\nSalaires : {salaires}")
print(f"Salaire moyen : {moyenne:.0f} EUR")

Exercice 5.4 : Manipuler du JSON [Difficulté : Intermédiaire]

Énoncé :

  1. Creez un dictionnaire Python representant un produit (nom, prix, stock, categories)
  2. Ecrivez-le dans un fichier produit.json avec json.dump() (indentation de 2)
  3. Relisez le fichier avec json.load() et affichez le resultat
  4. Convertissez un dictionnaire en chaine JSON avec json.dumps()
  5. Convertissez une chaine JSON en dictionnaire avec json.loads()

Indice : import json. json.dump(obj, f, indent=2) ecrit dans un fichier. json.load(f) lit depuis un fichier. json.dumps() et json.loads() travaillent avec des chaines.

Solution 
import json

# 1. Creer le dictionnaire
produit = {
    "nom": "Laptop Pro",
    "prix": 1299.99,
    "stock": 42,
    "categories": ["informatique", "bureautique"],
    "specs": {
        "ram": "16 Go",
        "stockage": "512 Go SSD",
        "ecran": "14 pouces"
    }
}

# 2. Ecrire en JSON
with open("produit.json", "w") as f:
    json.dump(produit, f, indent=2, ensure_ascii=False)
print("Fichier produit.json cree.")

# 3. Relire le JSON
with open("produit.json", "r") as f:
    data = json.load(f)
print(f"\nType : {type(data)}")
print(f"Nom  : {data['nom']}")
print(f"Prix : {data['prix']} EUR")
print(f"RAM  : {data['specs']['ram']}")

# 4. dumps : dict -> str
chaine_json = json.dumps(produit, indent=2, ensure_ascii=False)
print(f"\njson.dumps() :\n{chaine_json}")

# 5. loads : str -> dict
texte = '{"ville": "Paris", "population": 2161000, "pays": "France"}'
ville = json.loads(texte)
print(f"\njson.loads() : {ville}")
print(f"Type : {type(ville)}")
print(f"Population : {ville['population']:,}")

Exercice 6.2 : NumPy — Opérations vectorisées [Difficulté : Intermédiaire]

Énoncé :

  1. Creez un array a = np.array([10, 20, 30, 40, 50]) et effectuez : a + 5, a * 2, a ** 2
  2. Creez un array b = np.array([1, 2, 3, 4, 5]) et calculez a + b, a * b, a / b
  3. Utilisez les comparaisons : a > 25, et l'indexation booleenne a[a > 25]
  4. Calculez les statistiques : mean(), std(), min(), max(), sum(), cumsum()
  5. Generez 1000 nombres aleatoires (distribution normale) et calculez moyenne et ecart-type

Indice : NumPy applique les operations element par element (vectorisation). L'indexation booleenne permet de filtrer avec un masque. np.random.randn(n) genere une distribution normale.

Solution 
import numpy as np

# 1. Operations scalaires
a = np.array([10, 20, 30, 40, 50])
print(f"a       = {a}")
print(f"a + 5   = {a + 5}")
print(f"a * 2   = {a * 2}")
print(f"a ** 2  = {a ** 2}")

# 2. Operations entre arrays
b = np.array([1, 2, 3, 4, 5])
print(f"\na + b = {a + b}")
print(f"a * b = {a * b}")
print(f"a / b = {a / b}")

# 3. Comparaisons et indexation booleenne
masque = a > 25
print(f"\na > 25       = {masque}")
print(f"a[a > 25]    = {a[masque]}")
print(f"a[a % 20 == 0] = {a[a % 20 == 0]}")

# 4. Statistiques
print(f"\nmean = {a.mean()}")
print(f"std  = {a.std():.2f}")
print(f"min  = {a.min()}, max = {a.max()}")
print(f"sum  = {a.sum()}")
print(f"cumsum = {a.cumsum()}")

# 5. Distribution normale
np.random.seed(42)
donnees = np.random.randn(1000)
print(f"\n1000 valeurs normales :")
print(f"  Moyenne    : {donnees.mean():.4f}")
print(f"  Ecart-type : {donnees.std():.4f}")
print(f"  Min        : {donnees.min():.4f}")
print(f"  Max        : {donnees.max():.4f}")

import pandas as pd
import seaborn as sns

tips = sns.load_dataset("tips")

# 1. Moyenne par jour
print("--- Tip moyen par jour ---")
print(tips.groupby("day")["tip"].mean().round(2))

# 2. Agregations multiples
print("\n--- Agregations par jour ---")
stats = tips.groupby("day").agg(
    tip_mean=("tip", "mean"),
    tip_sum=("tip", "sum"),
    bill_mean=("total_bill", "mean"),
    count=("total_bill", "count"),
).round(2)
print(stats)

# 3. Groupby multiple
print("\n--- Moyenne par day + time ---")
multi = tips.groupby(["day", "time"])["total_bill"].mean().round(2)
print(multi)

# 4. Transform
tips["tip_moy_jour"] = tips.groupby("day")["tip"].transform("mean").round(2)
tips["diff_moy"] = (tips["tip"] - tips["tip_moy_jour"]).round(2)
print("\n--- Transform ---")
print(tips[["day", "tip", "tip_moy_jour", "diff_moy"]].head(10))

# 5. Pivot table
print("\n--- Pivot table ---")
pivot = pd.pivot_table(
    tips,
    values="tip",
    index="day",
    columns="time",
    aggfunc="mean",
).round(2)
print(pivot)

# Bonus : pivot avec marges
print("\n--- Pivot avec marges ---")
pivot_m = pd.pivot_table(
    tips,

).round(2)

Indice : ax.fill_between(x, y-std, y+std, alpha=0.2) cree une zone. ax.annotate("texte", xy, xytext, arrowprops) ajoute une annotation avec fleche.

Solution 
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

tips = sns.load_dataset("tips")

# Calculs
stats = tips.groupby("size")["tip"].agg(["mean", "std", "count"]).reset_index()
stats.columns = ["size", "tip_mean", "tip_std", "count"]

fig, ax = plt.subplots(figsize=(10, 6))

# Ligne principale
ax.plot(stats["size"], stats["tip_mean"], "o-", color="#1565C0",
        linewidth=2.5, markersize=8, label="Pourboire moyen", zorder=3)

# Zone ecart-type
ax.fill_between(
    stats["size"],
    stats["tip_mean"] - stats["tip_std"],
    stats["tip_mean"] + stats["tip_std"],
    alpha=0.2, color="#42A5F5", label="Ecart-type (+/- 1 sd)"
)

# Annotation du maximum
idx_max = stats["tip_mean"].idxmax()
x_max = stats.loc[idx_max, "size"]
y_max = stats.loc[idx_max, "tip_mean"]
ax.annotate(
    f"Max : {y_max:.2f} $\n(table de {x_max})",
    xy=(x_max, y_max),
    xytext=(x_max + 0.8, y_max + 0.5),
    fontsize=10,
    arrowprops=dict(arrowstyle="->", color="#333"),
    bbox=dict(boxstyle="round,pad=0.3", facecolor="#FFF9C4", edgecolor="#F9A825"),
)

# Ajouter le nombre d'observations
for _, row in stats.iterrows():
    ax.text(row["size"], row["tip_mean"] - 0.35, f"n={int(row['count'])}",
            ha="center", fontsize=8, color="gray")

# Style
ax.set_xlabel("Taille de la table (personnes)", fontsize=12)
ax.set_ylabel("Pourboire moyen ($)", fontsize=12)
ax.set_title("Pourboire moyen selon la taille de la table", fontsize=14, fontweight="bold")
ax.set_xticks(stats["size"])
ax.grid(True, alpha=0.3, linestyle="--")
ax.legend(loc="upper left", framealpha=0.9)
ax.set_xlim(0.5, stats["size"].max() + 0.5)

plt.tight_layout()
plt.savefig("graphique_pro.png", dpi=200, bbox_inches="tight")
plt.show()
