Lire un fichier texte

open, read et readlines

# Lire tout le contenu d'un coup
f = open('notes.txt', 'r')
contenu = f.read()
print(contenu)
f.close()

# Lire ligne par ligne dans une liste
f = open('notes.txt', 'r')
lignes = f.readlines()
for ligne in lignes:
    print(ligne.strip())
f.close()

Le bloc with

Le gardien qui ferme la porte

# Le gardien qui ferme la porte derriere vous
with open('notes.txt', 'r') as f:
    contenu = f.read()
    print(contenu)
# Ici, f est automatiquement ferme

# Lire ligne par ligne (economise la memoire)
with open('notes.txt', 'r') as f:
    for ligne in f:

Écrire dans un fichier

Modes w et a

# Écrire (écrase le fichier existant)
with open('resultat.txt', 'w') as f:
    f.write('Ligne 1\n')
    f.write('Ligne 2\n')

# Ajouter à la fin du fichier
with open('resultat.txt', 'a') as f:
    f.write('Ligne 3 ajoutee\n')

# Ecrire plusieurs lignes d'un coup
lignes = ['Alice: 18\n', 'Bob: 15\n', 'Clara: 20\n']
with open('notes_eleves.txt', 'w') as f:
    f.writelines(lignes)

Le module csv

csv.reader et DictReader

import csv

# csv.reader : chaque ligne est une liste
with open('eleves.csv', 'r') as f:
    lecteur = csv.reader(f)
    for ligne in lecteur:
        print(ligne)  # ['Alice', '18', 'Paris']

# csv.DictReader : chaque ligne est un dict
with open('eleves.csv', 'r') as f:
    lecteur = csv.DictReader(f)
    for eleve in lecteur:
        print(eleve['nom'], eleve['note'])

Le module json

load, dump, loads, dumps

import json

# Lire un fichier JSON
with open('config.json', 'r') as f:
    data = json.load(f)
    print(data['nom'])  # accès comme un dict

# Ecrire un fichier JSON
resultat = {'score': 95, 'eleve': 'Alice'}
with open('sortie.json', 'w') as f:
    json.dump(resultat, f, indent=2)

# Depuis/vers une chaine (le 's' = string)
texte = json.dumps(resultat)
obj = json.loads(texte)

Chemins et dossiers

os et pathlib

import os
from pathlib import Path

# Verifier si un fichier existe
print(os.path.exists('notes.txt'))  # True/False

# Lister les fichiers d'un dossier
for nom in os.listdir('.'):
    print(nom)

# pathlib : la methode moderne
chemin = Path('data') / 'notes.txt'
print(chemin.exists())
print(chemin.suffix)   # .txt
print(chemin.stem)     # notes

# Creer un dossier s'il n'existe pas
Path('resultats').mkdir(exist_ok=True)

Gestion d'erreurs

try / except / finally

try:
    with open("inexistant.txt", "r", encoding="utf-8") as f:
        contenu = f.read()
except FileNotFoundError:
    print("Fichier introuvable")
finally:
    print("Bloc terminé")
  • Capturer l'erreur la plus précise possible
  • Éviter les except: trop larges
  • Journaliser les cas d'échec importants
