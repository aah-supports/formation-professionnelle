import pandas as pd

df = pd.DataFrame({
    'nom': ['Alice', 'Bob', 'Clara'],
    'note': [18, 12, 15],
    'ville': ['Paris', 'Lyon', 'Paris']
})

# loc : acces par etiquette
print(df.loc[0, 'nom'])        # 'Alice'
print(df.loc[0:1, ['nom', 'note']])

# iloc : acces par position
print(df.iloc[0, 0])           # 'Alice'
print(df.iloc[:2, :2])         # 2 premieres lignes, 2 colonnes

# Selectionner une colonne

Filtrer les données

Conditions et masques booléens

import pandas as pd

df = pd.DataFrame({
    'nom': ['Alice', 'Bob', 'Clara', 'David'],
    'note': [18, 12, 15, 9],
    'ville': ['Paris', 'Lyon', 'Paris', 'Lille']
})

# Filtre simple
bons = df[df['note'] >= 14]
print(bons)  # Alice, Clara

# Conditions combinees (& = et, | = ou)
resultat = df[(df['note'] >= 14) & (df['ville'] == 'Paris')]
print(resultat)  # Alice, Clara

# Filtrer avec isin
grandes_villes = df[df['ville'].isin(['Paris', 'Lyon'])]

import pandas as pd

df = pd.DataFrame({
    'nom': ['Alice', 'Bob', 'Clara', 'David'],
    'note': [18, 12, 15, 9],
    'ville': ['Paris', 'Lyon', 'Paris', 'Lille']
})

# Tri par une colonne (croissant par defaut)
print(df.sort_values('note'))

# Tri decroissant
print(df.sort_values('note', ascending=False))

# Tri multi-colonnes
print(df.sort_values(['ville', 'note'], ascending=[True, False]))

# Remettre l'index en ordre
df_filtre = df[df['note'] >= 12]
print(df_filtre.sort_index())

Fusionner des DataFrames

merge et concat

import pandas as pd

# Deux tables a fusionner
df_notes = pd.DataFrame({
    'nom': ['Alice', 'Bob', 'Clara'],
    'note': [18, 12, 15]
})
df_villes = pd.DataFrame({
    'nom': ['Alice', 'Bob', 'David'],
    'ville': ['Paris', 'Lyon', 'Lille']
})

# merge : jointure sur la colonne 'nom'
resultat = pd.merge(df_notes, df_villes, on='nom', how='inner')
print(resultat)  # Alice et Bob seulement

# left join : garder toutes les lignes de gauche
resultat_left = pd.merge(df_notes, df_villes, on='nom', how='left')

# concat : empiler des DataFrames
df_all = pd.concat([df_notes, df_notes], ignore_index=True)

Grouper et agréger

groupby et agrégation

import pandas as pd

df = pd.DataFrame({
    'ville': ['Paris', 'Lyon', 'Paris', 'Lyon', 'Paris'],
    'ventes': [100, 80, 120, 90, 110],
    'cout': [50, 40, 60, 45, 55]
})

# Grouper par ville, calculer la moyenne
print(df.groupby('ville')['ventes'].mean())
# Lyon     85.0
# Paris   110.0

# Plusieurs agregations a la fois
stats = df.groupby('ville').agg(
    total_ventes=('ventes', 'sum'),
    moyenne_cout=('cout', 'mean'),
    nb_lignes=('ventes', 'count')
)
print(stats)
