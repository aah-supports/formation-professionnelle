Étape 3 : Nettoyage

Préparer les données pour l'analyse

# Créer la colonne pourcentage de pourboire
df['tip_pct'] = (df['tip'] / df['total_bill'] * 100).round(1)

# Détecter les valeurs aberrantes
print('Pourboires extrêmes :')
print(df[df['tip_pct'] > 40][['total_bill', 'tip', 'tip_pct']])

# Recoder en français pour les graphiques
jours_fr = {'Sun': 'Dim', 'Sat': 'Sam',
            'Thur': 'Jeu', 'Fri': 'Ven'}
df['jour'] = df['day'].map(jours_fr)

moment_fr = {'Lunch': 'Déjeuner', 'Dinner': 'Dîner'}
df['moment'] = df['time'].map(moment_fr)

print(df[['jour', 'moment', 'tip_pct']].head())

Étape 4 : Statistiques

Répondre aux questions avec les chiffres

# Pourboire moyen par jour
par_jour = df.groupby('jour')['tip_pct'].agg(
    ['mean', 'median', 'count']
).round(1)
print('Par jour :\n', par_jour)

# Déjeuner vs Dîner
par_moment = df.groupby('moment')['tip_pct'].mean().round(1)
print(f'\nDéjeuner : {par_moment["Déjeuner"]}%')
print(f'Dîner : {par_moment["Dîner"]}%')

# Fumeurs vs non-fumeurs
par_fumeur = df.groupby('smoker')['tip_pct'].mean().round(1)
print(f'\nFumeurs : {par_fumeur["Yes"]}%')
print(f'Non-fumeurs : {par_fumeur["No"]}%')

# Corrélation addition / pourboire
corr = df['total_bill'].corr(df['tip'])
print(f'\nCorrélation addition-pourboire : {corr:.2f}')

Étape 5 : Visualisation

Transformer les chiffres en histoire visuelle

fig, axes = plt.subplots(2, 2, figsize=(14, 10))

# 1. Scatter : addition vs pourboire
axes[0, 0].scatter(df['total_bill'], df['tip'],
                   alpha=0.6, c='#00f3ff', s=40)
axes[0, 0].set_title('Addition vs Pourboire')
axes[0, 0].set_xlabel('Addition ($)')
axes[0, 0].set_ylabel('Pourboire ($)')

# 2. Boxplot par jour
jours_ordre = ['Jeu', 'Ven', 'Sam', 'Dim']
data_box = [df[df['jour'] == j]['tip_pct'] for j in jours_ordre]
axes[0, 1].boxplot(data_box, labels=jours_ordre, patch_artist=True)
axes[0, 1].set_title('Pourboire (%) par jour')

# 3. Histogramme du pourcentage
axes[1, 0].hist(df['tip_pct'], bins=25, color='#10b981',
                edgecolor='white')
axes[1, 0].set_title('Distribution du pourboire (%)')

# 4. Barplot par moment
moyennes = df.groupby('moment')['tip_pct'].mean()
axes[1, 1].bar(moyennes.index, moyennes.values,
               color=['#a78bfa', '#ff6b00'])
axes[1, 1].set_title('Pourboire moyen par moment')

plt.tight_layout()
plt.savefig('dashboard_tips.png', dpi=150)
plt.show()
