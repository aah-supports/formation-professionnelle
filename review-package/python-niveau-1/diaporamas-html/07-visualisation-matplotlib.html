Nuage de points

plt.scatter

import matplotlib.pyplot as plt
import numpy as np

# Generer des donnees
np.random.seed(42)
ages = np.random.randint(18, 65, 50)
salaires = ages * 500 + np.random.normal(0, 5000, 50)

# Nuage de points
plt.figure(figsize=(8, 5))
plt.scatter(ages, salaires, c='#a78bfa', alpha=0.7, s=60)
plt.title('Salaire en fonction de l\'age')
plt.xlabel('Age')
plt.ylabel('Salaire (euros)')
plt.grid(True, alpha=0.3)
plt.show()

Histogramme

plt.hist pour les distributions

import matplotlib.pyplot as plt
import numpy as np

# Simuler des notes d'examen
np.random.seed(42)
notes = np.random.normal(12, 3, 200)
notes = np.clip(notes, 0, 20)  # borner entre 0 et 20

plt.figure(figsize=(8, 5))
plt.hist(notes, bins=20, color='#10b981',
         edgecolor='white', alpha=0.8)
plt.axvline(notes.mean(), color='#ff6b00',
            linestyle='--', label=f'Moyenne: {notes.mean():.1f}')
plt.title('Distribution des notes')
plt.xlabel('Note')
plt.ylabel('Nombre d\'eleves')
plt.legend()
plt.show()

Boîte à moustaches

plt.boxplot pour comparer

import matplotlib.pyplot as plt
import numpy as np

np.random.seed(42)
maths = np.random.normal(13, 3, 100)
physique = np.random.normal(11, 4, 100)
francais = np.random.normal(14, 2, 100)

plt.figure(figsize=(8, 5))
bp = plt.boxplot([maths, physique, francais],
                 labels=['Maths', 'Physique', 'Francais'],
                 patch_artist=True)
colors = ['#00f3ff', '#a78bfa', '#ff6b00']
for patch, color in zip(bp['boxes'], colors):
    patch.set_facecolor(color)
plt.title('Comparaison des notes par matiere')
plt.ylabel('Note')
plt.grid(True, axis='y', alpha=0.3)
plt.show()

Diagramme en barres

plt.bar et plt.barh

import matplotlib.pyplot as plt
import numpy as np

villes = ['Paris', 'Lyon', 'Marseille', 'Toulouse', 'Bordeaux']
ventes = [320, 180, 210, 150, 170]
colors = ['#00f3ff', '#a78bfa', '#ff6b00', '#10b981', '#a78bfa']

plt.figure(figsize=(8, 5))
bars = plt.bar(villes, ventes, color=colors, edgecolor='white')

# Ajouter les valeurs au-dessus des barres
for bar, v in zip(bars, ventes):
    plt.text(bar.get_x() + bar.get_width()/2, v + 5,
             str(v), ha='center', fontweight='bold')

plt.title('Ventes par ville')
plt.ylabel('Ventes (k euros)')
plt.show()

Personnalisation et export

Styles, annotations et savefig

import matplotlib.pyplot as plt
import numpy as np

# Utiliser un style predéfini
plt.style.use('seaborn-v0_8-darkgrid')

x = np.linspace(0, 10, 100)
y = np.sin(x)

fig, ax = plt.subplots(figsize=(10, 5))
ax.plot(x, y, color='#00f3ff', linewidth=2)

# Annoter un point
ax.annotate('Maximum', xy=(np.pi/2, 1),
            xytext=(np.pi/2 + 1, 0.8),
            arrowprops=dict(arrowstyle='->', color='#ff6b00'),
            fontsize=12, color='#ff6b00')

ax.set_title('Sinusoide annotee', fontsize=16)

# Exporter en haute resolution
fig.savefig('graphique.png', dpi=150, bbox_inches='tight')
