# ============================================================
# Projet integrateur — Analyse des pourboires
# Python Niveau I
# ============================================================

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# -----------------------------------------------------------
# Etape 1 : Import et exploration
# -----------------------------------------------------------

tips = sns.load_dataset("tips")

print("=== EXPLORATION ===")
print(f"Dimensions : {tips.shape[0]} lignes x {tips.shape[1]} colonnes\n")

print("--- Premieres lignes ---")
print(tips.head())

print("\n--- Dernieres lignes ---")
print(tips.tail())

print("\n--- Info ---")
tips.info()

print("\n--- Statistiques ---")
print(tips.describe())

print("\n--- Valeurs uniques ---")
for col in tips.select_dtypes(include="category").columns:
    print(f"  {col} : {tips[col].unique().tolist()}")

# -----------------------------------------------------------
# Etape 2 : Nettoyage et preparation
# -----------------------------------------------------------

print("\n=== NETTOYAGE ===")

# Pourcentage de pourboire
tips["tip_pct"] = (tips["tip"] / tips["total_bill"] * 100).round(1)

# Categorisation des additions
def categoriser_bill(montant):
    if montant < 15:
        return "Petit"
    elif montant < 30:
        return "Moyen"
    else:
        return "Gros"

tips["bill_cat"] = tips["total_bill"].apply(categoriser_bill)

# Traduction des jours
jours_fr = {"Sun": "Dimanche", "Sat": "Samedi", "Thur": "Jeudi", "Fri": "Vendredi"}
tips["jour_fr"] = tips["day"].map(jours_fr)

print(tips[["total_bill", "tip", "tip_pct", "bill_cat", "jour_fr"]].head(10))
print(f"\nRepartition bill_cat :\n{tips['bill_cat'].value_counts()}")

# -----------------------------------------------------------
# Etape 3 : Statistiques descriptives
# -----------------------------------------------------------

print("\n=== STATISTIQUES ===")

# Pourboire moyen par jour
print("--- Tip moyen par jour ---")
print(tips.groupby("jour_fr")["tip"].mean().round(2))

# Pourboire moyen par sexe
print("\n--- Tip moyen par sexe ---")
print(tips.groupby("sex")["tip"].mean().round(2))

# Pourboire moyen par jour et moment
print("\n--- Tip moyen par jour + moment ---")
print(tips.groupby(["jour_fr", "time"])["tip"].mean().round(2))

# Correlation
corr = tips["total_bill"].corr(tips["tip"])
print(f"\nCorrelation total_bill / tip : {corr:.4f}")

# Tableau croise
print("\n--- Crosstab jour x fumeur ---")
ct = pd.crosstab(tips["jour_fr"], tips["smoker"], margins=True)
print(ct)

# -----------------------------------------------------------
# Etape 4 : Visualisations
# -----------------------------------------------------------

fig, axes = plt.subplots(2, 2, figsize=(14, 10))
fig.suptitle("Analyse des pourboires — Restaurant", fontsize=16, fontweight="bold")

# 1. Scatter : total_bill vs tip
ax = axes[0, 0]
couleurs_sex = {"Male": "#2196F3", "Female": "#E91E63"}
for sexe, grp in tips.groupby("sex"):
    ax.scatter(grp["total_bill"], grp["tip"], c=couleurs_sex[sexe],
               alpha=0.6, label=sexe, edgecolors="white", linewidth=0.5)

coeffs = np.polyfit(tips["total_bill"], tips["tip"], 1)
x_line = np.linspace(0, 55, 100)
ax.plot(x_line, np.poly1d(coeffs)(x_line), "k--", linewidth=1.5, label="Regression")
ax.set_xlabel("Addition ($)")
ax.set_ylabel("Pourboire ($)")
ax.set_title("Addition vs Pourboire")
ax.legend(fontsize=9)

# 2. Histogramme : distribution de tip_pct
ax = axes[0, 1]
ax.hist(tips["tip_pct"], bins=25, color="#66BB6A", edgecolor="white", alpha=0.8)
moy_pct = tips["tip_pct"].mean()
med_pct = tips["tip_pct"].median()
ax.axvline(moy_pct, color="red", linestyle="--", linewidth=2,
           label=f"Moyenne : {moy_pct:.1f}%")
ax.axvline(med_pct, color="blue", linestyle="-.", linewidth=2,
           label=f"Mediane : {med_pct:.1f}%")
ax.set_xlabel("Pourboire (%)")
ax.set_ylabel("Frequence")
ax.set_title("Distribution du % de pourboire")
ax.legend(fontsize=9)

# 3. Boxplot : tip par jour
ax = axes[1, 0]
jours_ordre = ["Thur", "Fri", "Sat", "Sun"]
data_box = [tips[tips["day"] == j]["tip"].values for j in jours_ordre]
bp = ax.boxplot(data_box, labels=["Jeu", "Ven", "Sam", "Dim"],
                patch_artist=True, medianprops=dict(color="red", linewidth=2))
couleurs_box = ["#BBDEFB", "#C8E6C9", "#FFE0B2", "#F8BBD0"]
for patch, c in zip(bp["boxes"], couleurs_box):
    patch.set_facecolor(c)
ax.set_ylabel("Pourboire ($)")
ax.set_title("Pourboire par jour")

# 4. Barplot groupe : tip moyen par jour + time
ax = axes[1, 1]
pivot = tips.groupby(["day", "time"])["tip"].mean().unstack(fill_value=0)
pivot = pivot.reindex(jours_ordre)
x = np.arange(len(jours_ordre))
w = 0.35

b1 = ax.bar(x - w / 2, pivot.get("Lunch", [0] * 4), w,
            label="Dejeuner", color="#42A5F5")
b2 = ax.bar(x + w / 2, pivot.get("Dinner", [0] * 4), w,
            label="Diner", color="#EF5350")
for bars in [b1, b2]:
    for bar in bars:
        h = bar.get_height()
        if h > 0:
            ax.text(bar.get_x() + bar.get_width() / 2, h + 0.05,
                    f"{h:.2f}", ha="center", fontsize=8)
ax.set_xticks(x)
ax.set_xticklabels(["Jeu", "Ven", "Sam", "Dim"])
ax.set_ylabel("Pourboire moyen ($)")
ax.set_title("Pourboire moyen par jour et moment")
ax.legend(fontsize=9)

plt.tight_layout()
plt.savefig("dashboard_tips.png", dpi=200, bbox_inches="tight")
plt.show()
print("\nDashboard exporte : dashboard_tips.png")
