Les facteurs

Variables catégorielles

taille <- c("M", "L", "S", "M", "L", "XL")
f <- factor(taille)
print(f)
# [1] M  L  S  M  L  XL
# Levels: L M S XL

levels(f)  # "L" "M" "S" "XL"

# Facteur ordonné
factor(taille,
  # imposer un ordre
  levels = c("S","M","L","XL"),
  ordered = TRUE)
