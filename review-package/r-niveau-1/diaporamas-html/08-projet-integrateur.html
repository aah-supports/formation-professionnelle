Étape 2 : Nettoyage

Préparer les données

# Retirer les NA d'ozone
aq <- airquality[!is.na(airquality$Ozone), ]
nrow(aq)  # 116 lignes restantes

# Mois en texte
mois_noms <- c("Mai","Juin","Juil","Août","Sept")
aq$Mois_txt <- mois_noms[aq$Month - 4]

# Catégorie température
aq$Cat_temp <- ifelse(aq$Temp > 80, "Chaud", "Tempéré")

Étape 3 : Statistiques

Résumer par groupe

# Moyenne d'ozone par mois
aggregate(Ozone ~ Mois_txt, data = aq, FUN = mean)

# Corrélation température / ozone
cor(aq$Temp, aq$Ozone)  # ~0.70

# Résumé par catégorie
aggregate(Ozone ~ Cat_temp, data = aq,
  FUN = function(x) c(moy = mean(x), et = sd(x)))

Étape 4 : Visualisations

Quatre graphiques clés

par(mfrow = c(2, 2))

# 1. Scatter
plot(aq$Temp, aq$Ozone, pch = 19, col = "#00f3ff",
  main = "Temp vs Ozone", xlab = "Temp (F)", ylab = "Ozone (ppb)")
abline(lm(Ozone ~ Temp, data = aq), col = "red", lwd = 2)

# 2. Histogramme
hist(aq$Ozone, breaks = 15, col = "#a78bfa",
  main = "Distribution Ozone", xlab = "Ozone")

# 3. Boxplot par mois
boxplot(Ozone ~ Month, data = aq, col = "#10b981",
  main = "Ozone par mois", xlab = "Mois", ylab = "Ozone")

# 4. Barplot
barplot(table(aq$Mois_txt), col = "#ff6b00",
