Exercice 3 : Fonctions

Conversion Celsius / Fahrenheit

celsius_to_f <- function(c) c * 9/5 + 32

celsius_to_f(100)  # 212
celsius_to_f(0)    # 32

temps_c <- c(0, 15, 25, 37, 100)
celsius_to_f(temps_c)
