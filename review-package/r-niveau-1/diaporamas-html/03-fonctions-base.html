Créer ses propres fonctions

Automatiser vos traitements

z_score <- function(x) {
  (x - mean(x)) / sd(x)
}

notes <- c(12, 15, 8, 18, 14)
z_score(notes)
# [1] -0.42  0.42 -1.54  1.26  0.28

# Argument par défaut
arrondir <- function(x, n = 2) {
  round(x, n)
}
arrondir(3.14159)     # 3.14
arrondir(3.14159, 3)  # 3.142
