# Ex 3.5 : Créer une fonction IMC
imc <- function(poids_kg, taille_m) {
  poids_kg / taille_m^2
}
imc(70, 1.75)  # 22.86

# Ex 3.6 : Fonction catégorielle
categorie_imc <- function(poids, taille) {
  val <- imc(poids, taille)
  ifelse(val < 18.5, "Insuffisance",
    ifelse(val < 25, "Normal",
      ifelse(val < 30, "Surpoids", "Obésité")))
}
categorie_imc(c(50, 70, 90, 120), c(1.70, 1.75, 1.70, 1.80))

# Ex 6.1 : Sélection
data(mtcars)
mtcars[, c("mpg", "cyl", "hp")]

# Ex 6.2 : Filtrage
mtcars[mtcars$mpg > 25, ]
mtcars[mtcars$cyl == 4 & mtcars$hp > 80, ]
subset(mtcars, am == 1 & gear >= 4, select = c(mpg, hp, gear))

# Ex 6.3 : Tri
mtcars[order(mtcars$mpg), ][1:5, ]           # 5 pires
mtcars[order(-mtcars$mpg), ][1:5, ]          # 5 meilleurs
mtcars[order(mtcars$cyl, -mtcars$mpg), ]     # par cyl puis mpg desc

# Ex 6.4 : Colonnes calculées
df <- mtcars
df$kml <- df$mpg * 0.425144  # conversion en km/L
df$puiss_poids <- df$hp / df$wt
df$eco <- ifelse(df$mpg > 20, "Économique", "Gourmand")

# Ex 6.5 : Merge
constructeurs <- data.frame(
  cyl = c(4, 6, 8),
  desc = c("Compact", "Intermédiaire", "Puissant")
)
merged <- merge(df, constructeurs, by = "cyl", all.x = TRUE)

# Ex 6.6 : Agrégation
aggregate(mpg ~ cyl, data = mtcars, FUN = mean)
aggregate(cbind(mpg, hp, wt) ~ cyl, data = mtcars, FUN = mean)
aggregate(mpg ~ cyl + am, data = mtcars, FUN = mean)

# Ex 6.7 : Pipeline complet
# Voitures manuelles, 4 ou 6 cylindres, triées par mpg décroissant
result <- mtcars[mtcars$am == 1 & mtcars$cyl %in% c(4, 6), ]
result <- result[order(-result$mpg), ]
result$eff <- result$mpg / result$hp
result[, c("mpg", "cyl", "hp", "wt", "eff")]
