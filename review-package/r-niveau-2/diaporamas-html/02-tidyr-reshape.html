Problèmes fréquents des exports Excel

  • Entêtes sur plusieurs lignes et cellules fusionnées
  • Colonnes "montant" au format texte
  • Feuilles multiples avec structures différentes
  • Lignes de total ou commentaires à supprimer

Lecture maîtrisée d'un classeur

library(readxl)

excel_sheets("objectifs.xlsx")

objectifs <- read_excel(
  "objectifs.xlsx",
  sheet = "Objectifs",
  skip = 2,
  col_types = c("text", "text", "numeric")

Standardiser les colonnes

library(janitor)
library(dplyr)

objectifs_clean <- objectifs |>
  clean_names() |>
  rename(
    region = zone_vente,
    objectif_ca = ca_cible
  ) |>
  filter(!is.na(region), region != "")

Passer du large au long

library(tidyr)

objectifs_tidy <- objectifs_wide |>
  pivot_longer(
    cols = jan_2025:dec_2025,
    names_to = "periode",
    values_to = "objectif_ca"

  separate(periode, c("mois", "annee"), sep = "_")

Import multi-fichiers

library(purrr)

imports <- list.files("exports", pattern = "\\.xlsx$", full.names = TRUE) |>
  set_names() |>
  map(~ read_excel(.x, sheet = "Data") |> clean_names()) |>
  list_rbind(names_to = "source_fichier")
6 / 10

Gestion d'erreur robuste

safe_read_excel <- purrr::possibly(
  ~ read_excel(.x, sheet = "Data") |> clean_names(),
  otherwise = NULL
)

resultats <- fichiers |> map(safe_read_excel)
valides <- resultats |> compact()

Checklist qualité avant jointure

  • Clés métier sans NA critiques
  • Volume attendu vs volume importé
