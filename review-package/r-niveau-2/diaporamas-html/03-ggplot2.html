Anti-join pour détecter les trous

produits_absents <- ventes_tbl |>
  distinct(produit_id) |>
  anti_join(catalogue_tbl |> distinct(produit_id), by = "produit_id")

clients_sans_vente <- clients_tbl |>
  anti_join(ventes_tbl |> distinct(client_id), by = "client_id")

Segmentation avec case_when()

ventes_segmentees <- base_analyse |>
  mutate(
    segment_panier = case_when(
      montant_commande >= 1000 ~ "Premium",
      montant_commande >= 300  ~ "Standard",
      TRUE ~ "Essentiel"
    )
  )

Top N par groupe

top_clients <- base_analyse |>
  group_by(region, client_id) |>
  summarise(ca = sum(montant_commande, na.rm = TRUE), .groups = "drop") |>
  group_by(region) |>
  mutate(rang = min_rank(desc(ca))) |>
  filter(rang <= 5)
