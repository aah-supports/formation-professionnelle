Fonctions de fenêtre (évolution)

evo <- ventes_tbl |>
  mutate(mois = substr(date, 1, 7)) |>
  group_by(region, mois) |>
  summarise(ca = sum(prix_net, na.rm = TRUE), .groups = "drop") |>
  group_by(region) |>
  arrange(mois, .by_group = TRUE) |>
  mutate(
    ca_mois_prec = lag(ca),
    var_mom = (ca - ca_mois_prec) / ca_mois_prec
  )

Fonction paramétrée

extraire_kpi <- function(con, date_debut, date_fin, region = NULL) {
  req <- tbl(con, "ventes") |>
    filter(date >= date_debut, date <= date_fin)

  if (!is.null(region)) req <- req |> filter(region == !!region)

  req |>
    summarise(ca_total = sum(prix_net, na.rm = TRUE),
              nb_commandes = n_distinct(id_commande)) |>
    collect()
}

safe_extract <- purrr::possibly(
  ~ extraire_kpi(con, .x$date_debut, .x$date_fin, .x$region),
  otherwise = tibble::tibble(ca_total = NA_real_, nb_commandes = NA_integer_)
)
